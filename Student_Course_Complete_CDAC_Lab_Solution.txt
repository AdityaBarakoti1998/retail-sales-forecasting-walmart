Student and Course Entity - Complete CDAC Lab Solution

Question:
Create Student and Course entities.

Relationship:
One Student -> Many Courses

Requirements:
1. Save Student
2. Save Courses
3. Display all courses of a given student

=====================================

Student Entity

@Entity
@Getter
@Setter
@NoArgsConstructor
@AllArgsConstructor
public class Student {

   @Id
   @GeneratedValue(strategy = GenerationType.IDENTITY)
   private Long studentId;

   private String name;

   private String email;

   @OneToMany(cascade = CascadeType.ALL)
   @JoinColumn(name = "student_id")
   private List<Course> courses;
}

=====================================

Course Entity

@Entity
@Getter
@Setter
@NoArgsConstructor
@AllArgsConstructor
public class Course {

   @Id
   @GeneratedValue(strategy = GenerationType.IDENTITY)
   private Long courseId;

   private String courseName;

   private double fees;
}

=====================================

Student Repository

public interface StudentRepository
       extends JpaRepository<Student, Long> {
}

=====================================

Service Interface

public interface StudentService {

   Student saveStudent(Student student);

   List<Course> getCoursesByStudent(Long studentId);
}

=====================================

Service Implementation

@Service
public class StudentServiceImpl
       implements StudentService {

   @Autowired
   private StudentRepository repository;

   @Override
   public Student saveStudent(Student student) {
       return repository.save(student);
   }

   @Override
   public List<Course> getCoursesByStudent(Long studentId) {

       Student student =
               repository.findById(studentId).orElse(null);

       return student.getCourses();
   }
}

=====================================

Controller

@RestController
@RequestMapping("/students")
public class StudentController {

   @Autowired
   private StudentService service;

   @PostMapping
   public Student saveStudent(
           @RequestBody Student student) {

       return service.saveStudent(student);
   }

   @GetMapping("/{id}/courses")
   public List<Course> getCourses(
           @PathVariable Long id) {

       return service.getCoursesByStudent(id);
   }
}
