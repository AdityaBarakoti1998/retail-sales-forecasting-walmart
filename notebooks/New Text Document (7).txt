package com.example.securitydemo.config;

import org.springframework.context.annotation.*;
import org.springframework.security.config.annotation.web.builders.HttpSecurity;
import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
import org.springframework.security.core.userdetails.*;
import org.springframework.security.provisioning.InMemoryUserDetailsManager;
import org.springframework.security.web.SecurityFilterChain;
import org.springframework.security.config.Customizer;
import org.springframework.http.HttpMethod;

@Configuration
@EnableWebSecurity
public class SecurityConfig {

    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        http
            .csrf(csrf -> csrf.disable())
            .authorizeHttpRequests(auth -> auth
                .requestMatchers(HttpMethod.GET, "/employees/**").hasAnyRole("ADMIN","USER")
                .requestMatchers(HttpMethod.POST, "/employees/**").hasRole("ADMIN")
                .requestMatchers(HttpMethod.PUT, "/employees/**").hasRole("ADMIN")
                .requestMatchers(HttpMethod.DELETE, "/employees/**").hasRole("ADMIN")
                .requestMatchers("/public/**").permitAll()
                .anyRequest().authenticated()
            )
            .httpBasic(Customizer.withDefaults());
        return http.build();
    }

    @Bean
    public UserDetailsService userDetailsService() {
        UserDetails admin = User.withDefaultPasswordEncoder()
            .username("admin").password("admin123").roles("ADMIN").build();
        UserDetails user = User.withDefaultPasswordEncoder()
            .username("user").password("user123").roles("USER").build();
        return new InMemoryUserDetailsManager(admin, user);
    }
}

EmployeeController.java — com.example.securitydemo.controller package

package com.example.securitydemo.controller;

import com.example.securitydemo.model.Employee;
import org.springframework.web.bind.annotation.*;
import java.util.*;

@RestController
@RequestMapping("/employees")
public class EmployeeController {

    private List<Employee> employees = new ArrayList<>(Arrays.asList(
        new Employee(1, "Alice", 50000),
        new Employee(2, "Bob", 60000)
    ));

    @GetMapping
    public List<Employee> getAll() { return employees; }

    @GetMapping("/{id}")
    public Employee getById(@PathVariable int id) {
        return employees.stream().filter(e -> e.getId() == id).findFirst().orElse(null);
    }

    @PostMapping
    public String add(@RequestBody Employee emp) {
        employees.add(emp); return "Employee added!";
    }

    @PutMapping("/{id}")
    public String update(@PathVariable int id, @RequestBody Employee emp) {
        for (Employee e : employees) {
            if (e.getId() == id) {
                e.setName(emp.getName()); e.setSalary(emp.getSalary()); return "Updated!";
            }
        }
        return "Not found!";
    }

    @DeleteMapping("/{id}")
    public String delete(@PathVariable int id) {
        employees.removeIf(e -> e.getId() == id); return "Deleted!";
    }
}

@RestController
public class StudentController {

    @Autowired
    private RestTemplate restTemplate;

    @GetMapping("/student/course/{id}")
    public String getStudentWithCourse(@PathVariable int id) {
        String url = "http://course-service/courses/" + id;
        String course = restTemplate.getForObject(url, String.class);
        return "Student enrolled in: " + course;
    }

    @GetMapping("/student/all-courses")
    public String getAllCourses() {
        String url = "http://course-service/courses";
        String courses = restTemplate.getForObject(url, String.class);
        return "All Courses: " + courses;
    }
}
